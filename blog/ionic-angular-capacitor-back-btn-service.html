Ionic Angular + Capacitor Back Button Service

A custom back button service for Ionic Angular applications using Capacitor.

In this blog post, we will create a custom back button service for Ionic Angular applications using Capacitor. This service will allow us to handle the hardware back button on Android devices and provide a consistent user experience across different platforms.

Creating the Back Button Service

export class AppComponent implements AfterViewInit {
    constructor(private backButtonService: BackButtonService) {}

    ngAfterViewInit(): void {
        this.backButtonService.registerRouterOutlet();

}
            
src/app/services/back-button.service.ts:
import { Injectable } from '@angular/core';
import { NavigationEnd, Router } from '@angular/router';
import { App } from '@capacitor/app';
import {
  ActionSheetController,
  AlertController,
  LoadingController,
  ModalController,
  Platform,
  PopoverController,
} from '@ionic/angular/standalone';
import { filter } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class BackButtonService {
  private initialized = false;
  private history: string[] = [];
  private isGoingBack = false;

  constructor(
    private platform: Platform,
    private router: Router,
    private modalController: ModalController,
    private alertController: AlertController,
    private loadingController: LoadingController,
    private actionSheetController: ActionSheetController,
    private popoverController: PopoverController,
  ) {}

  registerRouterOutlet(): void {
    if (!this.initialized) {
      this.initialize();
      this.initialized = true;
    }
  }

  private initialize(): void {
    this.history = [this.router.url];
    this.router.events
      .pipe(
        filter(
          (event): event is NavigationEnd => event instanceof NavigationEnd,
        ),
      )
      .subscribe((event) => {
        if (this.isGoingBack) {
          this.isGoingBack = false;
          return;
        }

        const url = event.urlAfterRedirects;
        const last = this.history[this.history.length - 1];
        if (last !== url) {
          this.history.push(url);
        }
      });

    this.platform.backButton.subscribeWithPriority(10000, () => {
      void this.handleBackButton();
    });
  }

  private async handleBackButton(): Promise {
    const loading = await this.loadingController.getTop();
    if (loading && this.hasNoBackButtonClass(loading)) {
      return;
    }

    const activeOverlay = await this.getActiveOverlay();
    if (activeOverlay) {
      await activeOverlay.dismiss();
      return;
    }

    console.error('history', this.history);
    if (this.history.length > 1) {
      this.isGoingBack = true;
      this.history.pop();
      const target = this.history[this.history.length - 1];
      console.error('target', target);
      if (target === '/') {
        await App.exitApp();
        return;
      }
      await this.router.navigateByUrl(target, { replaceUrl: true });
      return;
    }

    await App.exitApp();
  }

  private async getActiveOverlay(): Promise<
    | HTMLIonModalElement
    | HTMLIonAlertElement
    | HTMLIonLoadingElement
    | HTMLIonActionSheetElement
    | HTMLIonPopoverElement
    | undefined
  > {
    return (
      (await this.modalController.getTop()) ??
      (await this.popoverController.getTop()) ??
      (await this.alertController.getTop()) ??
      (await this.actionSheetController.getTop()) ??
      (await this.loadingController.getTop()) ??
      undefined
    );
  }

  private hasNoBackButtonClass(loading: HTMLIonLoadingElement): boolean {
    return (loading as HTMLElement).classList.contains('no-back-button');
  }
}

MainActivity.java:

  package io.ionic.starter;
  
  import android.os.Bundle;
  import android.view.WindowManager;
  
  import com.getcapacitor.BridgeActivity;
  
  public class MainActivity extends BridgeActivity {
  
  @Override
  public void onCreate(Bundle savedInstanceState) {
  super.onCreate(savedInstanceState);
  getWindow().setBackgroundDrawableResource(android.R.color.transparent);
  this.clearFlags();
  }
  
  @Override
  public void onUserLeaveHint() {
  super.onUserLeaveHint();
  this.addFlags();
  }
  
  @Override
  public void onPause() {
  super.onPause();
  this.addFlags();
  }
  
  @Override
  public void onWindowFocusChanged(boolean hasFocus) {
  super.onWindowFocusChanged(hasFocus);
  
  if (!hasFocus) {
  this.addFlags();
  } else {
  this.clearFlags();
  }
  }
  
  @Override
  public void onResume() {
  super.onResume();
  this.clearFlags();
  }
  
  private void addFlags() {
  getWindow().addFlags(WindowManager.LayoutParams.FLAG_SECURE);
  }
  
  private void clearFlags() {
  getWindow().clearFlags(WindowManager.LayoutParams.FLAG_SECURE);
  }
  }
